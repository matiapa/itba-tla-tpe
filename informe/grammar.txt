
A -> + | - | * | / | % | ^

R -> < | > | <= | >= | eq | neq

L -> and | or | not

type -> number | text | boolean

variable -> [a-zA-Z_][a-zA-Z0-9_]*

list -> type variable [ number ]

equation -> variable A variable

exp -> ( ( number R number ) M ) | ( ( boolean ) M ) | ( boolean )
     | ( number R number )

M -> L exp M | L exp 

I -> if exp do \n 

W -> while exp do \n

# supongo que lines termina con un \n
T -> I lines else \n lines end \n | I lines end \n

V -> W lines end \n

F -> mean | median | mode | stdev | range | qtr1 | qtr3 | inter-qtr | gcd | mcm

opeartion -> F of list | number comb number | number perm number | number !

asign -> type variable \n | type variable = arg \n | type variable = opeartion \n
